# 大小端问题

+ 大端模式：将高位字节放低地址位，低位字节放高位地址
+ 小端模式：将低位字节放低地址位，高位字节放高位地址
**网络字节序都是大端模式**
+ 大小端转换
~~~ c
// 函数实现
uint16_t ntohs(uint16_t target) // network to host (short)
uint16_t htons(uint16_t target)
uint32_t ntohl(uint32_t tartget) // network to host (long)
uint32_t htonl(uint32_t target)

// 位运算
(short)((target & 0x00FF) << 8 | (target & 0xFF00 >> 8))

(int)((target & 0x000000FF) << 24 | (target & 0x0000FF00) << 8 | (target & 0x00FF0000) >> 8 | (target & 0xFF000000) >> 24)
~~~
+ 判断大小端
~~~ c
// 方法一
union {
	int a;
	char b;
}test;
test.a = 1;
if(test.b == 1) 小端
else 大端

// 方法二
int i = 1;
char* p = (char*)&i;
if(*p == 1) 小端
else 大端
~~~

## 冒泡排序

+ 每一轮循环比较相邻元素，把最大（最小）的数冒出

~~~c++
void bubbleSort(vector<int> &v){
    int len = v.size();
    for (int i = 0; i < len; i++){
        for(int j = 0;j < len - 1 - i;j++){
            if (arr[j] > arr[j + 1]) {
                int tmp = arr[j + 1];
                arr[j + 1] = arr[j];
                arr[j] = tmp;
            }
        }
    }  
}
~~~

## 选择排序

+ 每次轮询把未排序的里面最大（最小）的选择出来，插到已排序后面

~~~c++
void selectSort(int arr[], int len) {
    int min_index;
    for (int i = 0; i < len; i++) {
        min_index = i;
        for (int j = i+1; j < len; j++) {
            if (arr[min_index] > arr[j]) {
                min_index = j;
            }
        }
        int tmp = arr[min_index];
        arr[min_index] = arr[i];
        arr[i] = tmp;
    }
}
~~~

## 插入排序
+ 将从已排序中往前遍历并依次比较未排序元素，将其插入合适位置

~~~c++
void insertSort(int arr[], int len) {
    int sort_index;
    int tmp;
    for (int i = 1; i < len; i++) {
        sort_index = i - 1;
        tmp = arr[i];
        while (sort_index >= 0 && arr[sort_index] > tmp) {
            arr[sort_index + 1] = arr[sort_index];
            sort_index--;
        }
        arr[sort_index + 1] = tmp;
    }
}
~~~

## 希尔排序
+ 分成若干子序列进行插入排序

~~~c++
void shellSort(int arr[], int len) {
    int h = 1;
    while (h < len / 3) {
        h = 3 * h + 1;
    }
    while (h >= 1) {
        for (int i = h; i < len; i++) {
            for (int j = i; j >= h && arr[j] < arr[j - h]; j -= h) {
                int tmp = arr[j];
                arr[j] = arr[j-h];
                arr[j-h] = tmp;
            }
        }
        h = h / 3;
    }
}
~~~

## 归并排序

+ 分治：将序列一直对半分，最后合并排序

~~~c++
void merge(int arr[], int begin, int mid, int end) {

    vector<int> tmp;
    for (int i = begin; i < mid; i++) {
        tmp.push_back(arr[i]);
    }
    int lb = begin, rb = mid, rd = end, index = 0;

    while (index < mid - begin) {
        if (rb < rd && tmp[index] > arr[rb]) {
            arr[lb++] = arr[rb++];
        } else {
            arr[lb++] = tmp[index++];
            
        }
    }

}

void mergeSort(int arr[], int begin, int end) {

    if ((end -begin) < 2) {
        return;
    }
    int mid = (end + begin) >> 1;
    mergeSort(arr, begin, mid);
    mergeSort(arr, mid, end);
    merge(arr, begin, mid, end);
    
}

void mergeSort(int arr[], int len) {
    mergeSort(arr, 0, len);
}
~~~

## 快速排序
+ 选择序列第一个元素为基准，将序列中比基准小的放左边，比基准大的放右边
+ 分好的左右序列递归调用重复上面的步骤，直到剩一个元素

~~~c++
int partition(int arr[], int begin, int end) {

    int pivot = arr[begin];

    while (begin < end) {
        while (begin < end && arr[end] > pivot) {
            end--;
        }
        arr[begin] = arr[end];
        while (begin < end && arr[begin] < pivot) {
            begin++;
        }
        arr[end] = arr[begin];
    }
    arr[begin] = pivot;
    return begin;

}

void quickSort(int arr[], int begin, int end) {
    if ((end - begin) < 2) return;

    int mid = partition(arr, begin, end);

    quickSort(arr, begin, mid - 1);
    quickSort(arr, mid + 1, end);
}

void quickSort(int arr[], int len) {
    quickSort(arr, 0, len-1);
}
~~~
